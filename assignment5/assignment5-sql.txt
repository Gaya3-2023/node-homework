   FROM products p JOIN line_items l on p.product_id=l.product_id 
                JOIN orders o on o.order_id=l.order_id

#Task 2 : Understanding Subqueries
SELECT c.customer_name,avg(t.total_price) as average_order_price 
    FROM customers C JOIN (SELECT ct.customer_id,o.order_id as order_id,sum(p.price*l.quantity) as total_price 
                            FROM products p JOIN line_items l on p.product_id=l.product_id 
                                JOIN orders o on o.order_id=l.order_id 
                                JOIN customers ct ON ct.customer_id=o.customer_id 
                                GROUP BY O.order_id,ct.customer_id) as T 
                        ON c.customer_id=T.customer_id 
                        GROUP BY c.customer_id,c.customer_name 
                        ORDER BY c.customer_name;

#Task 4 - Aggregation with Having
select e.first_name,e.last_name,count(o.order_id) as order_count 
from orders o JOIN employees e ON e.employee_id= o.employee_id 
group by e.employee_id,e.first_name,e.last_name 
Having count(o.order_id)>5 order by e.last_name;
